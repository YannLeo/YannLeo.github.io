    N,V = map(int, input().split())

v = [0] * (N+1)
w = [0] * (N+1)
s = [0] * (N+1)

for i in range(1, N+1):
    v[i], w[i], s[i] = map(int, input().split())


f = [0] *(V+1)

for i in range(1, N+1):
    for j in range(V,v[i]-1,-1):
        for k in range(1, min(s[i], j//v[i])+1):
            f[j] = max(f[j], f[j-k*v[i]]+k*w[i])

print(f[V])
